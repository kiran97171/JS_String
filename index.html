const countCharacters = (S) => 
{
  let countOfA=0 , countOfD=0;
  for(let i=0; i < S.length; i++){
    if(S[i]=="A")
       countOfA++;
    if(S[i]=="D")
       countOfD++;
  }
  let result=[];
  result.push(countOfA);
  result.push(countOfD);
  return result;
};

Que8:Plaindrome Check

Ans:

var Palin_Check = (str) =>
{
let count = 0;
let len = str.length - 1;
for(let i = 0; i < str.length ; i++){
if(str[i] === str[len]){
count++;
len--;
}else{
return "False";
}
}
if(count>0){
return "True";
}	 
}
